AUTOSSH_POLL=10 \
AUTOSSH_FIRST_POLL=10 \
AUTOSSH_GATETIME=0 \
autossh -f \
  -M 0 \
  -N \
  -o "TCPKeepAlive=yes" \
  -o "ServerAliveInterval=10" \
  -o "ServerAliveCountMax=1" \
  -o "ExitOnForwardFailure=yes" \
  -o "Compression=no" \
  -o "ControlMaster=no" \
  -o "ControlPersist=no" \
  -p [port for ssh] \
  -i ~/.ssh/[path to key] \
  -L 0.0.0.0:[port servers]:127.0.0.1:[portservers] \
  [login]@[ip]
